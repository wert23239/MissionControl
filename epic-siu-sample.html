Epic SIU feed → schedule slot cache sample

Sketch for using daily Epic availability pulls plus HL7 SIU appointment events to keep cached openings fresh.

How hard is this?

Medium. Not algorithmically hard, but integration-hard. The core code is straightforward: parse SIU messages, identify appointment IDs / provider / department / start time, and remove or update matching cached slots. The harder parts are customer-specific HL7 mappings, feed transport, Epic environment quirks, retries/idempotency, and reconciliation when messages are late/missing.

  • Prototype: 1–3 days if you have sample SIU messages.
  • Production MVP: 1–3 weeks depending on feed setup, customer testing, and mapping variance.
  • Robust multi-customer version: 4–8+ weeks because every site’s SIU feed can differ.

Architecture

  • Nightly/off-hours job calls GetScheduleDaysForProvider2 for all in-scope provider/department/visit-type combinations, 60 days out.
  • Normalize openings into a cache keyed by providerId + departmentId + startDateTime + visitTypeId.
  • Group slots by visit type so the UI/search layer can answer quickly.
  • HL7 SIU feed receives appointment creates/updates/cancels.
  • For booked/new appointment events, remove matching open slots from cache. For cancels, mark for reconciliation or re-query that provider/day.

Sample Node.js code

// epic-siu-slot-cache-sample.js
// Raw demo only: no Epic API client, no MLLP listener, no DB adapter.
// Purpose: parse HL7 SIU messages and update an in-memory availability cache.

const crypto = require('crypto');

// ----- Slot cache -----------------------------------------------------------

class SlotCache {
  constructor() {
    // slotKey -> slot
    this.slots = new Map();
    // visitTypeId -> Set(slotKey)
    this.byVisitType = new Map();
    // appointmentId -> slotKey, if Epic/HL7 gives a reliable appointment id
    this.byAppointmentId = new Map();
  }

  upsertSlot(slot) {
    const key = makeSlotKey(slot);
    const normalized = { ...slot, key, status: slot.status || 'open', updatedAt: new Date().toISOString() };

    this.slots.set(key, normalized);

    for (const visitTypeId of normalized.visitTypeIds || [normalized.visitTypeId]) {
      if (!visitTypeId) continue;
      if (!this.byVisitType.has(visitTypeId)) this.byVisitType.set(visitTypeId, new Set());
      this.byVisitType.get(visitTypeId).add(key);
    }

    if (normalized.appointmentId) this.byAppointmentId.set(normalized.appointmentId, key);
    return normalized;
  }

  removeSlot(slotKey, reason) {
    const slot = this.slots.get(slotKey);
    if (!slot) return false;

    this.slots.delete(slotKey);

    for (const visitTypeId of slot.visitTypeIds || [slot.visitTypeId]) {
      const set = this.byVisitType.get(visitTypeId);
      if (set) set.delete(slotKey);
    }

    if (slot.appointmentId) this.byAppointmentId.delete(slot.appointmentId);
    console.log('removed slot', { slotKey, reason });
    return true;
  }

  removeByAppointmentId(appointmentId, reason) {
    const slotKey = this.byAppointmentId.get(appointmentId);
    if (!slotKey) return false;
    return this.removeSlot(slotKey, reason);
  }

  findLikelySlot({ providerId, departmentId, startDateTime }) {
    for (const [key, slot] of this.slots.entries()) {
      if (
        slot.providerId === providerId &&
        slot.departmentId === departmentId &&
        sameMinute(slot.startDateTime, startDateTime)
      ) {
        return { key, slot };
      }
    }
    return null;
  }

  listByVisitType(visitTypeId) {
    return [...(this.byVisitType.get(visitTypeId) || [])]
      .map(key => this.slots.get(key))
      .filter(Boolean)
      .sort((a, b) => new Date(a.startDateTime) - new Date(b.startDateTime));
  }
}

function makeSlotKey(slot) {
  return [slot.providerId, slot.departmentId, slot.startDateTime, slot.visitTypeId || (slot.visitTypeIds || []).join(',')]
    .map(v => String(v || '').trim())
    .join('|');
}

function sameMinute(a, b) {
  return Math.floor(new Date(a).getTime() / 60000) === Math.floor(new Date(b).getTime() / 60000);
}

// ----- Daily Epic availability cache refresh --------------------------------

async function refreshDailyAvailability({ providers, departments, visitTypes, days = 60, epicClient, cache }) {
  for (const providerId of providers) {
    for (const departmentId of departments) {
      for (const visitTypeId of visitTypes) {
        // Placeholder for Epic GetScheduleDaysForProvider2.
        // Run off-hours. Throttle aggressively. Persist progress/checkpoints.
        const openings = await epicClient.getScheduleDaysForProvider2({
          providerId,
          departmentId,
          visitTypeId,
          startDate: todayIsoDate(),
          endDate: addDaysIsoDate(days),
        });

        for (const opening of openings) {
          cache.upsertSlot({
            providerId,
            departmentId,
            visitTypeId,
            visitTypeIds: opening.allowedVisitTypeIds || [visitTypeId],
            startDateTime: opening.startDateTime,
            endDateTime: opening.endDateTime,
            source: 'GetScheduleDaysForProvider2',
          });
        }
      }
    }
  }
}

function todayIsoDate() {
  return new Date().toISOString().slice(0, 10);
}

function addDaysIsoDate(days) {
  const d = new Date();
  d.setUTCDate(d.getUTCDate() + days);
  return d.toISOString().slice(0, 10);
}

// ----- HL7 SIU parsing ------------------------------------------------------

function parseHl7(raw) {
  const normalized = raw.replace(/\n/g, '\r').split('\r').filter(Boolean);
  const segments = normalized.map(line => line.split('|'));
  const byName = new Map();

  for (const fields of segments) {
    const name = fields[0];
    if (!byName.has(name)) byName.set(name, []);
    byName.get(name).push(fields);
  }

  return {
    raw,
    segments,
    get(name, index = 0) {
      return byName.get(name)?.[index] || null;
    },
    all(name) {
      return byName.get(name) || [];
    },
  };
}

function parseSiuEvent(rawHl7) {
  const msg = parseHl7(rawHl7);
  const msh = msg.get('MSH');
  const sch = msg.get('SCH');
  const pv1 = msg.get('PV1');
  const rgs = msg.get('RGS');
  const ais = msg.get('AIS');
  const aip = msg.get('AIP');
  const ail = msg.get('AIL');

  if (!msh || !sch) throw new Error('Not enough HL7 data: expected MSH and SCH');

  const messageType = field(msh, 9);       // e.g. SIU^S12
  const triggerEvent = component(messageType, 2); // S12, S13, S14, S15, S17, etc.

  return {
    messageId: field(msh, 10),
    eventType: triggerEvent,
    eventMeaning: siuMeaning(triggerEvent),
    appointmentId: firstNonEmpty(
      entityId(field(sch, 1)), // placer appointment id
      entityId(field(sch, 2)), // filler appointment id
      field(sch, 26)           // site-specific fallback in some feeds
    ),
    appointmentReason: component(field(sch, 7), 2),
    appointmentType: component(field(sch, 8), 1),
    startDateTime: hl7TsToIso(firstNonEmpty(
      field(sch, 11),          // timing quantity can vary by feed
      field(ais, 4),
      field(aip, 6)
    )),
    durationMinutes: numberOrNull(firstNonEmpty(field(sch, 9), field(ais, 6), field(aip, 9))),
    patientClass: field(pv1, 2),
    departmentId: firstNonEmpty(
      locationId(field(ail, 3)),
      locationId(field(pv1, 3)),
      field(rgs, 3)
    ),
    providerId: firstNonEmpty(
      personId(field(aip, 3)),
      personId(field(pv1, 7)), // attending doctor, feed-dependent
      personId(field(pv1, 8))
    ),
    visitTypeId: firstNonEmpty(
      component(field(ais, 3), 1),
      component(field(sch, 8), 1)
    ),
  };
}

function siuMeaning(eventType) {
  return {
    S12: 'new appointment booking',
    S13: 'appointment rescheduling',
    S14: 'appointment modification',
    S15: 'appointment cancellation',
    S17: 'appointment deletion',
    S26: 'appointment no-show',
  }[eventType] || 'unknown/other SIU event';
}

function applySiuToCache(rawHl7, cache) {
  const event = parseSiuEvent(rawHl7);

  // The exact SIU events to treat as “slot taken” can vary by customer.
  const slotTakenEvents = new Set(['S12', 'S13', 'S14']);
  const slotReleasedEvents = new Set(['S15', 'S17']);

  if (slotTakenEvents.has(event.eventType)) {
    // Prefer exact appointment id if the cache has one; otherwise match by provider/dept/start.
    if (event.appointmentId && cache.removeByAppointmentId(event.appointmentId, event.eventMeaning)) {
      return { action: 'removed_by_appointment_id', event };
    }

    const match = cache.findLikelySlot(event);
    if (match) {
      cache.removeSlot(match.key, event.eventMeaning);
      return { action: 'removed_by_provider_department_time', event };
    }

    // Could be a booking for a slot we never cached, or mapping mismatch. Log for reconciliation.
    return { action: 'no_matching_slot_found_reconcile', event };
  }

  if (slotReleasedEvents.has(event.eventType)) {
    // Do not blindly add a slot back from SIU cancel/delete: visit-type rules and holds may have changed.
    // Safer: enqueue targeted re-query for provider/day via GetScheduleDaysForProvider2.
    return { action: 'enqueue_targeted_requery_for_provider_day', event };
  }

  return { action: 'ignored', event };
}

function field(segment, hl7FieldNumber) {
  if (!segment) return '';
  // Most segments: segment[0] is name, segment[1] is HL7 field 1.
  // MSH is special: MSH-1 is the field separator itself, so after split('|')
  // segment[1] is MSH-2, segment[8] is MSH-9, segment[9] is MSH-10, etc.
  if (segment[0] === 'MSH') {
    if (hl7FieldNumber === 1) return '|';
    return segment[hl7FieldNumber - 1] || '';
  }
  return segment[hl7FieldNumber] || '';
}

function component(value, componentNumber) {
  if (!value) return '';
  return String(value).split('^')[componentNumber - 1] || '';
}

function entityId(ei) {
  // EI: entity id ^ namespace id ^ universal id ^ universal id type
  return component(ei, 1) || '';
}

function personId(xcn) {
  // XCN: id ^ family ^ given ...
  return component(xcn, 1) || '';
}

function locationId(pl) {
  // PL often: point of care ^ room ^ bed ^ facility ...
  // Department might be in point of care/facility depending on site mapping.
  return component(pl, 1) || component(pl, 4) || '';
}

function hl7TsToIso(ts) {
  if (!ts) return null;
  // Handles basic YYYYMMDDHHmmss. Real production code should preserve timezone if supplied.
  const clean = String(ts).replace(/[^0-9+-]/g, '');
  const y = clean.slice(0, 4);
  const m = clean.slice(4, 6) || '01';
  const d = clean.slice(6, 8) || '01';
  const hh = clean.slice(8, 10) || '00';
  const mm = clean.slice(10, 12) || '00';
  const ss = clean.slice(12, 14) || '00';
  return `${y}-${m}-${d}T${hh}:${mm}:${ss}`;
}

function numberOrNull(value) {
  const n = Number(value);
  return Number.isFinite(n) ? n : null;
}

function firstNonEmpty(...values) {
  return values.find(v => v !== undefined && v !== null && String(v).trim() !== '') || '';
}

Production gotchas

  • SIU field mappings are not universal. Validate against each customer’s sample messages.
  • Never assume a cancel means the slot is immediately available; re-query the provider/day.
  • Make SIU handling idempotent using MSH-10/message ID plus appointment ID.
  • Track reconciliation drift: nightly full refresh + event feed during the day is the sane combo.
  • For Epic, the integration lift is often access, testing, and mapping — not the cache code.
